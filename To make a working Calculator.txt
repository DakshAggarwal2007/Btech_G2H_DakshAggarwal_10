#include <stdio.h>
#include <math.h>
int main() {
    float a,b,c;
    printf("Enter the two numbers:\n");
    scanf("%f%f",&a,&b);
    printf("Enter 1 for '+':\nEnter 2 for '-':\nEnter 3 for '*':\nEnter 4 for '/':\n");
    scanf("%f",&c);
    if(c==1){
        printf("%f",a+b);
    }
    else if(c==2){
        printf("%f",a-b);
    }
    else if(c==3){
        printf("%f",a*b);
    }
    else if(c==4){
        printf("%f",a/b);
    }
    else {
        printf("Enter a valid number");
    }
    return 0;
}
